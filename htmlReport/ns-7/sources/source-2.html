 inline fun <reified T: Any> jsonToObject(json: String) : T
         = jacksonObjectMapper()
     .registerModule(JodaModule())
     .configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false)
     .readValue<T>(json)

 }
